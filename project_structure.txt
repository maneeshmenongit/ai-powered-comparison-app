## Phase 1 Files (POC)
- main.py                        # Start here
- src/llm/intent_parser.py       # AI query parsing

- src/api_clients/lyft_client.py # Mock Lyft API

- src/api_clients/base_client.py # Shared API logic
- src/api_clients/adapters.py    # Data normalization
- src/services/cache_service.py  # API caching
- config.py                      # API credentials

- app.py                         # Streamlit app
- src/services/comparison_service.py # Business logic
- static/                        # CSS, images (if needed)

## Phase 4 Files (Intelligence)
- src/database/                  # All database files
- src/llm/memory.py             # Conversation memory
- src/services/recommendation_service.py # Personalization

# Quick Commands Reference

# Set up new project
mkdir comparison-app && cd comparison-app
python -m venv venv
source venv/bin/activate
pip install -r requirements.txt

# Run CLI version
python main.py

# Run web version
streamlit run app.py

# Run tests
pytest tests/

# Check code quality
flake8 src/
black src/  # Auto-format

# Git workflow
git init
git add .
git commit -m "Initial commit"
git remote add origin <your-repo>
git push -u origin main
